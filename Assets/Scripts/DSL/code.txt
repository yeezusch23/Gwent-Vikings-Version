effect {
    Name: "Damage",
    Params: {
        Amount: Number
    },
    Action: (targets, context) => {
        for (target in targets) {
            i = 0;
            while (i++ < Amount)
                target.Power -= 1;
        }
    }
}

effect {
    Name: "Draw",
    Action: (targets, context) => {
        topCard = context.Deck.Pop();
        context.Hand.Push(topCard);
        context.Hand.Shuffle();
    }
}
